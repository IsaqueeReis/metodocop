const fs = require('fs');
let code = fs.readFileSync('App.tsx', 'utf-8');

const target = "{formatTime(session.durationSeconds</span></div>)";
const replacement = "{formatTime(session.durationSeconds)}</span></div>))}";

if(code.includes(target)) {
    code = code.replace(target, replacement);
    fs.writeFileSync('App.tsx', code);
    console.log('Fixed durationSeconds map closure!');
} else {
    console.log('Target not found!');
}
